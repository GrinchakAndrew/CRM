    <script type="text/javascript">
        var spreadsheet = "https://docs.google.com/spreadsheets/d/1zBZN4DCd8TKs6JqhLp73UNGajWRwy0QNzCUtaKnLkZ4/edit#gid=0";
        $(document).ready(function() {
            $('#classes').sheetrock({
                url: spreadsheet,
                query: "select A, B, C, E where C is not null",
                labels: ['Tasks', 'Description', 'PHP'],
                callback: function(a, b, c) {
				var ctx = document.getElementById("myChart").getContext('2d');
				var ctx2 = document.getElementById("myChart2").getContext('2d');
				ctx.canvas.width = 300;
				ctx.canvas.height = 300;
				ctx2.canvas.width = 300;
				ctx2.canvas.height = 300;
					
					var timeout = setTimeout(function(){
					var data = {
						status_index : c.rows[0]['labels'].indexOf('status'),
						done : '',
						in_process : ''
					};
					data.done = c.rows.filter(function(i, j) {
						if(~i['cellsArray'][data.status_index].indexOf('done') || ~i['cellsArray'][data.status_index].indexOf('Done')) {
							return i['cellsArray'][data.status_index];
						 }
					});
					
					data.in_process = c.rows.filter(function(i, j) {
							if(~i['cellsArray'][data.status_index].indexOf('process')) {
								return i['cellsArray'][data.status_index];
							}
						});
					
					var tasksDone = data.done.length; 
					var tasksInProcess = data.in_process.length;
					var tasksIncomplete = c.rows.length - (tasksDone + tasksInProcess);
					
                    var myChart = new Chart(ctx, {
                        type: 'bar',
                        data: {
                            labels: ["Done", "In Process", "Incomplete"],
                            datasets: [{
                                label: ['# of Tasks'],
                                data: [tasksDone, tasksInProcess, tasksIncomplete],
                                backgroundColor: [
                                    'rgba(75, 192, 192, 0.2)',
									 'rgba(54, 162, 235, 0.2)',
									  'rgba(255, 206, 86, 0.2)'
                                ],
                                borderColor: [
                                    'rgba(75, 192, 192, 1)',
									 'rgba(54, 162, 235, 1)',
									 'rgba(255, 206, 86, 1)'
                                ],
                                borderWidth: 1
                            }]
                        },
                        options: {
							responsive: true,
							maintainAspectRatio: true,
                            scales: {
                                yAxes: [{
                                    ticks: {
                                        beginAtZero: true
                                    }
                                }]
                            }, 
							animation: {
								animateScale: true,
								animateRotate: true
							}
                        }
                    });
					
					var myChart2 = new Chart(ctx2, {
                        type: 'pie',
                        data: {
                            labels: ["Done", "In Process", "Incomplete"],
                            datasets: [{
                                label: '# of Tasks',
                                data: [tasksDone, tasksInProcess, tasksIncomplete],
                                backgroundColor: [
                                    'rgba(75, 192, 192, 0.2)',
									 'rgba(54, 162, 235, 0.2)',
									  'rgba(255, 206, 86, 0.2)'
                                ],
                                borderColor: [
                                    'rgba(75, 192, 192, 1)',
									 'rgba(54, 162, 235, 1)',
									 'rgba(255, 206, 86, 1)'
                                ],
                                borderWidth: 1
                            }]
                        },
                        options: {
							responsive: true,
							maintainAspectRatio: true,
							animation: {
								animateScale: true,
								animateRotate: true
							}, 
							pieceLabel: {
							mode: 'percentage',
							precision: 1,
							fontColor: '#000000', 
							arc: false
						  }
                        }
                    });
					
					}, 1000);
				}
            });
        });
    </script>
